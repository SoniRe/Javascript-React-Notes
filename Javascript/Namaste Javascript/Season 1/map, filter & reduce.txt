-> map() - It is used to Tranform the array

Eg - 

const arr = [5, 1, 3, 6];

function double(x) {
  return 2 * x;
}

function triple(x) {
  return 3 * x;
}

function binary(x) {
  return x.toString(2);
}

const output = arr.map(binary);

// Double - [10, 2, 6, 12]

// Triple - [15, 3, 9, 18]

// Binary - ['101', '1', '11', '110']

-> filter() - It is used to filter out values

// filter odd values
const isOdd = (val) => {
  return val % 2 !== 0;
};

function isEven(val) {
  return val % 2 === 0;
}

function greaterThan4(val) {
  return val > 4;
}

const output = arr.filter(greaterThan4);

console.log(output);

-> reduce() - Take all elements of array and come up with single value

Eg- 
// Sum of all elements
// Find maximum number inside the array

const arr = [5, 1, 3, 6];

// sum or max

function findSum(arr) {
  let sum = 0;

  for (let i = 0; i < arr.length; i++) {
    sum += arr[i];
  }

  return sum;
}

const sum = findSum(arr);

console.log(sum);

const output = arr.reduce(function (accumulator, current) {
  // accumulator - Accumulate the result
  // current - represents value in the array

  // Second value of reducer is initial value
  accumulator = accumulator + current;
  return accumulator;
}, 0);

console.log(output);
